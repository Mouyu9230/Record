## 给你一个大小为 m x n 的矩阵 grid 。最初，你位于左上角 (0, 0) ，每一步，你可以在矩阵中 向右 或 向下 移动。在从左上角 (0, 0) 开始到右下角 (m - 1, n - 1) 结束的所有路径中，找出具有 最大非负积 的路径。路径的积是沿路径访问的单元格中所有整数的乘积。返回 最大非负积 对 109 + 7 取余 的结果。如果最大积为 负数 ，则返回 -1 。注意，取余是在得到最大积之后执行的。

初窥dp的门径，解答如下

```
class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int m=grid.size();
        int n=grid[0].size();

        vector<vector<long long>> dp_max(m, vector<long long>(n));
        vector<vector<long long>> dp_min(m, vector<long long>(n));//用两个与grid相同大小的vector来存放“走到当前格时，可能情况中的最大/小值”，至于为什么需要俩，因为这道题测试数据全是负数，也就是说max可能来自前一格的min.

        dp_max[0][0]=grid[0][0];//初始化
        dp_min[0][0]=grid[0][0];

        for(int i=1;i<m;i++){//第一列，数据只能来自“上方”，直接一溜下来就能获取数据。既然只有一个可能值，当然maxmin都等于它
            dp_max[i][0]=dp_max[i-1][0]*grid[i][0];
            dp_min[i][0]=dp_min[i-1][0]*grid[i][0];

        }

        for(int i=1;i<n;i++){//第一行，同上
            dp_max[0][i]=dp_max[0][i-1]*grid[0][i];
            dp_min[0][i]=dp_min[0][i-1]*grid[0][i];

        }



        for(int i=1;i<m;i++){//从(1,1)开始
            for(int j=1;j<n;j++){
//所有可能性
                long long int a=dp_max[i-1][j]*grid[i][j];//来自上max
                long long int b=dp_min[i-1][j]*grid[i][j];//来自上min
                long long int c=dp_max[i][j-1]*grid[i][j];//来自左max
                long long int d=dp_min[i][j-1]*grid[i][j];//来自左min

                long long int MAX=max(max(a,b),max(c,d));//abcd里挑
                long long int MIN=min(min(a,b),min(c,d));

                dp_max[i][j]=MAX;
                dp_min[i][j]=MIN;

            }
        }

        long long int finalmax=dp_max[m-1][n-1];
        if(finalmax<0){
            return -1;
        }

        return finalmax%1000000007;

    }
};
```
